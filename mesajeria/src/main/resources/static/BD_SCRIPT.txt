CREATE TABLE VEHICULO(
	ID_VEHICULO INTEGER PRIMARY KEY AUTOINCREMENT,
	RESPONSABLE VARCHAR(30),
	LATITUD REAL DEFAULT 0,
	LONGITUD REAL DEFAULT 0
);

CREATE TABLE ESTADO(
	ID_ESTADO INTEGER PRIMARY KEY AUTOINCREMENT,
	ESTADO VARCHAR(30)
);

CREATE TABLE PEDIDO(
	ID_PEDIDO INTEGER PRIMARY KEY AUTOINCREMENT,
	ID_VEHICULO INTEGER,
	ID_ESTADO INTEGER,
	ID_USU INTEGER,
	HORA_PEDIDO DATE,
	PRODUCTO VARCHAR(30),
	FOREIGN KEY(ID_VEHICULO) REFERENCES VEHICULO,
	FOREIGN KEY(ID_ESTADO) REFERENCES ESTADO,
	FOREIGN KEY(ID_USU) REFERENCES USUARIOS
);

CREATE TABLE REGISTRO_UBICACION(
	ID_REGISTRO INTEGER PRIMARY KEY AUTOINCREMENT,
	ID_VEHICULO INTEGER,
	HORA_REGISTRO DATE,
	LATITUD REAL,
	LONGITUD REAL,
	FOREIGN KEY(ID_VEHICULO) REFERENCES VEHICULO
);

CREATE TABLE USUARIO(
	ID_USU INTEGER PRIMARY KEY AUTOINCREMENT,
	NOMBRE VARCHAR(30),
	EMAIL VARCHAR(30)
);
